  function boltPath(x1,y1,x2,y2,rough){
    let pts=[{x:x1,y:y1},{x:x2,y:y2}], off=rough;
    for(let it=0;it<6;it++){
      const np=[pts[0]];
      for(let i=1;i<pts.length;i++){
        const a=pts[i-1],b=pts[i];
        np.push({x:(a.x+b.x)/2+(Math.random()-0.5)*off,
                 y:(a.y+b.y)/2+(Math.random()-0.5)*off*0.3});
        np.push(b);
      }
      pts=np; off*=0.5;
    }
    return pts;
  }

      if(reduced || fall<=0 || fall>=0.98) return;
      const power=Math.min(1,0.18+speed/34+fall*0.35);
      const sourceY=y+bladeH*(0.22+0.45*fall);
      const count=2+Math.floor(power*5);
      for(let i=0;i<count;i++){
        const side=i%2?-1:1;
        windPuffs.push({
          x:cx+side*(12+Math.random()*28)+(Math.random()-0.5)*18,
          y:sourceY-Math.random()*bladeH*.35,
          vx:side*(0.45+Math.random()*1.25)+(Math.random()-0.5)*0.4,
          vy:-0.8-Math.random()*1.5-speed*0.025,
          r:10+Math.random()*30+power*18,
          life:0.72+Math.random()*0.28,
          spin:(Math.random()-0.5)*0.06,
          hue:Math.random()<0.7?'220,228,236':'179,77,255'
        });
      }
      for(let i=0;i<3+Math.floor(power*5);i++){
        const side=i%2?-1:1;
        windStreaks.push({
          x:cx+side*(20+Math.random()*42),
          y:sourceY-Math.random()*bladeH*.6,
          len:70+Math.random()*150+speed*3,
          vx:side*(1.2+Math.random()*2.8),
          vy:-2.2-Math.random()*3.4-speed*0.05,
          a:0.25+Math.random()*0.35,
          life:0.42+Math.random()*0.22,
          side
        });
      }
    }

    function drawWind(){
      wctx.clearRect(0,0,W,H);
      wctx.globalCompositeOperation='lighter';
      /* falling light trail behind the blade + hot tip — fades when scroll stops */
      if(trailGlow>0.02 && !reduced){
        const len=Math.min(H*0.55, 110+trailGlow*H*0.42);
        const y0=trailY+bladeH*0.15;
        const wid=9+trailGlow*16;
        const g1=wctx.createLinearGradient(0,y0-len,0,y0+bladeH*0.4);
        g1.addColorStop(0,'rgba(179,77,255,0)');
        g1.addColorStop(0.7,'rgba(196,204,232,'+(0.10*trailGlow)+')');
        g1.addColorStop(1,'rgba(238,242,255,'+(0.22*trailGlow)+')');
        wctx.fillStyle=g1;
        wctx.fillRect(cx-wid/2,y0-len,wid,len+bladeH*0.4);
        const tipY=trailY+bladeH*0.98;
        const rg=wctx.createRadialGradient(cx,tipY,0,cx,tipY,24+trailGlow*32);
        rg.addColorStop(0,'rgba(238,242,255,'+(0.32*trailGlow)+')');
        rg.addColorStop(1,'rgba(179,77,255,0)');
        wctx.fillStyle=rg;
        wctx.beginPath(); wctx.arc(cx,tipY,24+trailGlow*32,0,Math.PI*2); wctx.fill();
        trailGlow*=0.88;
      }
      for(let i=windPuffs.length-1;i>=0;i--){
        const p=windPuffs[i];
        p.life-=0.018; p.x+=p.vx; p.y+=p.vy; p.vx*=0.992; p.vy-=0.012; p.r+=0.42;
        if(p.life<=0){ windPuffs.splice(i,1); continue; }
        const g=wctx.createRadialGradient(p.x,p.y,0,p.x,p.y,p.r);
        g.addColorStop(0,'rgba('+p.hue+','+(0.18*p.life)+')');
        g.addColorStop(0.45,'rgba('+p.hue+','+(0.08*p.life)+')');
        g.addColorStop(1,'rgba('+p.hue+',0)');
        wctx.fillStyle=g;
        wctx.beginPath();
        wctx.ellipse(p.x,p.y,p.r*1.15,p.r*0.58,p.spin+p.vx*0.04,0,Math.PI*2);
        wctx.fill();
      }
      wctx.lineCap='round';
      for(let i=windStreaks.length-1;i>=0;i--){
        const s=windStreaks[i];
        s.life-=0.035; s.x+=s.vx; s.y+=s.vy; s.len*=0.985;
        if(s.life<=0){ windStreaks.splice(i,1); continue; }
        const grad=wctx.createLinearGradient(s.x,s.y,s.x-s.side*s.len,s.y+s.len*.18);
        grad.addColorStop(0,'rgba(238,244,255,'+(s.a*s.life)+')');
        grad.addColorStop(1,'rgba(179,77,255,0)');
        wctx.strokeStyle=grad;
        wctx.lineWidth=1.1+2.4*s.life;
        wctx.beginPath();
        wctx.moveTo(s.x,s.y);
        wctx.quadraticCurveTo(s.x-s.side*s.len*.42,s.y+s.len*.03,s.x-s.side*s.len,s.y+s.len*.18);
        wctx.stroke();
      }
      wctx.globalCompositeOperation='source-over';
      requestAnimationFrame(drawWind);
    }
    drawWind();

    let impacted=false, dropStorming=false;
    function thunder(){
      thunderSound(0.95,2.6);
      dropStorming=true;
      /* one channel hits the blade, a weaker one lands off to the side */
      const st=buildStrike(cx+(Math.random()-0.5)*60,-20,cx,groundY,W*0.14);
      const st2=buildStrike(cx+(Math.random()<0.5?-1:1)*W*0.3,-20,cx+(Math.random()-0.5)*W*0.2,groundY*0.92,W*0.11);
      strikeSeq(
        a=>{
          bctx.clearRect(0,0,W,H);
          drawStrike(bctx,st,2.6,a,W*0.42);
          drawStrike(bctx,st2,1.3,a*0.45,0);
        },
        a=>{ flash.style.opacity=(a*0.38).toFixed(2); },
        ()=>{ bctx.clearRect(0,0,W,H); flash.style.opacity=0; dropStorming=false; },
        true
      );
      if(window.gsap){
        gsap.fromTo(shock,{scale:0,opacity:0.9},{scale:30,opacity:0,duration:.95,ease:'power2.out'});
        gsap.fromTo(glow,{opacity:0},{opacity:1,duration:.16,yoyo:true,repeat:1,ease:'power1.out'});
      }
      if(!reduced) shake();
    }
    function shake(){
      let f=0; const max=16;
      (function sh(){
        f++; const damp=Math.max(0,1-f/max), k=12*damp;
        world.style.transform='translate('+((Math.random()-0.5)*k).toFixed(1)+'px,'+((Math.random()-0.5)*k).toFixed(1)+'px)';
        if(f<max) requestAnimationFrame(sh); else world.style.transform='translate(0,0)';
      })();
    }
    function resetThunder(){ bctx.clearRect(0,0,W,H); flash.style.opacity=0; shock.style.opacity=0; glow.style.opacity=0; }

    cut.style.opacity=0;
    cut.style.transform='translate(-50%,34px) scale(.94)';

    function setDropText(fall, vanish, appear, speed){
      const textPower=clamp((fall-0.08)/0.58,0,1);
      const impactPower=clamp((fall-0.76)/0.22,0,1);
      const fadeOut=clamp((appear-0.25)/0.65,0,1);
      const opacity=textPower*(1-fadeOut*0.72);
      glyphs.style.opacity=opacity.toFixed(3);
      glyphs.style.transform='translateY('+((1-textPower)*30-speed*0.15).toFixed(1)+'px)';
      for(let i=0;i<glyphItems.length;i++){
        const local=clamp((textPower-i*0.09)/0.82,0,1);
        glyphItems[i].style.opacity=(0.16+local*0.72).toFixed(3);
        glyphItems[i].style.transform='translateY('+((1-local)*42-speed*(i%2?0.08:-0.05)).toFixed(1)+'px)';
        glyphItems[i].style.filter='blur('+((1-local)*2.6).toFixed(2)+'px)';
      }
      const sealOpacity=clamp((impactPower*1.15)-fadeOut*0.8,0,1);
      seal.style.opacity=sealOpacity.toFixed(3);
      seal.style.transform='translate(-50%,-50%) scale('+(0.82+impactPower*0.24+speed*0.002).toFixed(3)+')';
      seal.style.filter='blur('+((1-impactPower)*2.8).toFixed(2)+'px)';
    }

    ScrollTrigger.create({
      trigger:'#drop', start:'top top', end:'bottom bottom', scrub:0.45,
      onUpdate(self){
        const p=self.progress;
        /* intro text: fully there at the start, blurs and lifts away as the fall begins */
const intro=1-clamp(p/0.16,0,1);
dropIntro.style.opacity=intro.toFixed(3);
dropIntro.style.transform='translate(-50%,-50%) translateY('+((1-intro)*-46).toFixed(1)+'px)';
dropIntro.style.filter='blur('+((1-intro)*9).toFixed(1)+'px)';
/* background portrait: soft at the start, blurs deeper and fades out by mid-fall */
const bgFade=1-clamp((p-0.08)/0.34,0,1);
dropBg.style.opacity=(0.55*bgFade).toFixed(3);
dropBg.style.filter='blur('+(2+(1-bgFade)*10).toFixed(1)+'px)';
        /* gravity fall: hangs at the top, accelerates, slams down */
        const t=clamp(p/0.64,0,1);
        const fall=Math.pow(t,1.7);
        const landingY=Math.max(H*0.46, H-bladeH-H*0.08);
        const y=landingY*fall;
        const speed=Math.abs(y-lastBladeY);
        /* slight air sway that settles as it gains speed */
        const sway=Math.sin(t*Math.PI*2.4)*4.5*(1-fall);
        const rotZ=180+sway;
        const sc=0.52+0.48*fall;
        /* speed smear: vertical stretch + motion blur, sharp again at rest */
        const stretch=1+Math.min(0.075,speed*0.003);
        const blur=Math.min(4.5,speed*0.11);
        const vanish=easeOut(clamp((p-0.64)/0.16,0,1));
        const appear=easeOut(clamp((p-0.70)/0.18,0,1));

        blade.style.opacity=(1-vanish).toFixed(3);
        blade.style.transform='translate(-50%,0) translateY('+y.toFixed(1)+'px) rotateX(180deg) rotateZ('+rotZ.toFixed(2)+'deg) scale('+sc.toFixed(3)+') scaleY('+stretch.toFixed(3)+')';
        blade.style.filter='blur('+blur.toFixed(2)+'px) drop-shadow(0 30px 44px rgba(0,0,0,.75)) drop-shadow(0 0 '+(30+speed*0.5).toFixed(0)+'px rgba(179,77,255,'+(0.36+Math.min(0.3,speed*0.01)).toFixed(2)+'))';
        trailY=y;
        trailGlow=Math.max(trailGlow,Math.min(1,speed/24));

        if(fall>0.02 && fall<0.99 && (fall>=lastFall || speed>2)) emitWind(y,fall,speed);
        lastBladeY=y; lastFall=fall;
        setDropText(fall, vanish, appear, speed);
        cut.style.opacity=appear.toFixed(3);
        cut.style.transform='translate(-50%,'+((1-appear)*34).toFixed(1)+'px) scale('+(0.94+0.06*appear).toFixed(3)+')';
        cut.style.filter='drop-shadow(0 34px 70px rgba(0,0,0,.78)) drop-shadow(0 0 '+(28+appear*40).toFixed(1)+'px rgba(179,77,255,.48))';
        glow.style.opacity=(appear*0.55).toFixed(3);
        cutPower=appear;

        if(p>=0.64 && !impacted){ impacted=true; thunder(); }
        if(p<0.58 && impacted){ impacted=false; resetThunder(); }
      }
    });

    /* ambient crackle around the revealed cut: real short-lived arcs
       spawned every few hundred ms, each fading over ~240ms —
       no per-frame random spaghetti */
    const aura=[]; let lastAura=0;
    (function thunderAura(){
      requestAnimationFrame(thunderAura);
      if(dropStorming) return; /* don't fight the impact strike frames */
      bctx.clearRect(0,0,W,H);
      if(cutPower>0.03 && !reduced){
        const now=performance.now();
        if(now-lastAura>260+Math.random()*440){
          lastAura=now;
          const side=Math.random()<0.5 ? W*(0.10+Math.random()*0.16) : W*(0.74+Math.random()*0.16);
          aura.push({
            st:buildStrike(side,H*(0.10+Math.random()*0.2), cx+(Math.random()-0.5)*W*0.24, H*(0.35+Math.random()*0.35), W*0.09),
            born:now, w:0.8+Math.random()*0.7
          });
          if(aura.length>4) aura.shift();
        }
        for(let i=aura.length-1;i>=0;i--){
          const s=aura[i], age=performance.now()-s.born;
          const a=cutPower*Math.max(0,1-age/240)*(age<36?age/36:1);
          if(a<=0){ aura.splice(i,1); continue; }
          drawStrike(bctx,s.st,s.w,a*0.75,0);
        }
      }
    })();
  })();

  /* ================= SLASH SPLIT HOVER ================= */
  const slashSec=document.getElementById('slash');
  const slashTop=document.getElementById('slashTop');
  const slashLine=document.getElementById('slashLine');
  const slashHint=document.getElementById('slashHint');
  const sparkC=document.getElementById('slashSpark');
  const sctx=sparkC.getContext('2d');
  const slashBoltC=document.getElementById('slashBolt');
  const slctx=slashBoltC.getContext('2d');
  const slashFlash=document.getElementById('slashFlash');
  const TILT=18*Math.PI/180;      // slash angle from vertical
  const TAN=Math.tan(TILT);

 let sx=0, sTarget=0, lastX=0, lastApplied=-1;
  let sparks=[];
  let sw=0, sh=0, halfShift=0;

  function resizeSpark(){
    sw=slashSec.clientWidth; sh=slashSec.clientHeight;
    halfShift=(sh/2)*TAN;
    sparkC.width=sw*DPR; sparkC.height=sh*DPR;
    sctx.setTransform(DPR,0,0,DPR,0,0);
    slashBoltC.width=sw*DPR; slashBoltC.height=sh*DPR;
    slctx.setTransform(DPR,0,0,DPR,0,0);
  }
  resizeSpark(); addEventListener('resize',resizeSpark);

  /* ---- full-screen thunder on hover ---- */
  let slashStorming=false, lastRumble=0;
  function slashThunder(intensity,focusX){
    if(reduced || slashStorming) return;
    slashStorming=true;
    const now=performance.now();
    if(now-lastRumble>1200){ thunderSound(0.7*intensity,1.8+intensity); lastRumble=now; }
    /* channels built once, restruck along the same path */
    const strikes=[];
    const mainX=(focusX!=null)? focusX+(Math.random()-0.5)*sw*0.1 : sw*(0.2+Math.random()*0.6);
    strikes.push({st:buildStrike(mainX,-20, mainX+(Math.random()-0.5)*sw*0.16, sh*(0.6+Math.random()*0.34), sw*0.13), w:2.4*Math.max(intensity,0.6), k:1});
    if(intensity>0.65){
      const x2=sw*(0.08+Math.random()*0.84);
      strikes.push({st:buildStrike(x2,-20, x2+(Math.random()-0.5)*sw*0.12, sh*(0.4+Math.random()*0.3), sw*0.1), w:1.3, k:0.5});
      /* cloud-to-cloud arc crawling near the top */
      strikes.push({st:buildStrike(sw*0.08, sh*(0.08+Math.random()*0.1), sw*0.92, sh*(0.1+Math.random()*0.14), sh*0.1), w:0.9, k:0.3});
    }
    strikeSeq(
      a=>{
        slctx.clearRect(0,0,sw,sh);
        strikes.forEach((s,i)=>drawStrike(slctx,s.st,s.w,a*s.k*intensity, i===0?sw*0.3:0));
      },
      a=>{ slashFlash.style.opacity=(a*intensity*0.3).toFixed(2); },
      ()=>{ slctx.clearRect(0,0,sw,sh); slashFlash.style.opacity=0; slashStorming=false; },
      intensity>0.7
    );
  }

  slashSec.addEventListener('mouseenter',()=>slashThunder(1));
  slashSec.addEventListener('touchstart',()=>slashThunder(1),{passive:true});

  let lastMoveBolt=0;
  function setPointer(clientX,clientY){
    const r=slashSec.getBoundingClientRect();
    const edge=sw ? halfShift/sw : 0;
    const raw=(clientX-r.left)/sw;
    sTarget=Math.min(1+edge,Math.max(-edge,raw*(1+edge*2)-edge));
    slashHint.style.opacity=0;
    const py=(clientY-r.top);
    const speed=Math.abs(sTarget-lastX);
    const n=Math.min(10,Math.floor(speed*160));
    for(let i=0;i<n;i++){
      sparks.push({
        x:sTarget*sw+(Math.random()-0.5)*10,
        y:py+(Math.random()-0.5)*80,
        vx:(Math.random()-0.5)*4,vy:(Math.random()-0.7)*3,
        r:0.8+Math.random()*2,life:1,
        col:Math.random()<0.5?'207,214,228':'179,77,255'
      });
    }
    /* fast blade drags call the storm down near the cursor */
    const now=performance.now();
    if(speed>0.05 && now-lastMoveBolt>800){
      lastMoveBolt=now;
      slashThunder(0.55+Math.min(0.45,speed*3), clientX-r.left);
    }
    lastX=sTarget;
  }
  slashSec.addEventListener('mousemove',e=>setPointer(e.clientX,e.clientY));
  slashSec.addEventListener('touchmove',e=>{const t=e.touches[0];setPointer(t.clientX,t.clientY);},{passive:true});
  slashSec.addEventListener('mouseleave',()=>{sTarget=sx;});

  (function slashFrame(){
    // fast direct lerp — responsive, no spring wobble
    sx += (sTarget - sx) * 0.28;

    // only touch the DOM when the change is visible
    if(Math.abs(sx-lastApplied) > 0.0004){
      const cx=sx*sw;
      const xTop=cx - halfShift;
      const xBot=cx + halfShift;
      slashTop.style.clipPath='polygon(0 0,'+xTop.toFixed(1)+'px 0,'+xBot.toFixed(1)+'px '+sh+'px,0 '+sh+'px)';
      slashLine.style.transform='translateX('+cx.toFixed(1)+'px) rotate(18deg)';
      const glow=Math.min(1,Math.abs(sTarget-sx)*6+0.4);
      slashLine.style.opacity=glow.toFixed(2);
      lastApplied=sx;
    }

    // sparks
    if(sparks.length){
      sctx.clearRect(0,0,sw,sh);
      sctx.globalCompositeOperation='lighter';
      for(let i=sparks.length-1;i>=0;i--){
        const p=sparks[i];
        p.x+=p.vx;p.y+=p.vy;p.vy+=0.12;p.life-=0.03;
        if(p.life<=0){sparks.splice(i,1);continue;}
        sctx.beginPath();sctx.arc(p.x,p.y,p.r*p.life,0,Math.PI*2);
        sctx.fillStyle='rgba('+p.col+','+p.life+')';sctx.fill();
      }
      sctx.globalCompositeOperation='source-over';
      if(!sparks.length) sctx.clearRect(0,0,sw,sh);
    }
    requestAnimationFrame(slashFrame);
  })();

  /* hud state by section */
  const hudState=document.getElementById('hudState');
  ScrollTrigger.create({trigger:'#eye',start:'top center',onEnter(){hudState.textContent='BLADES: SENSING';},onLeaveBack(){hudState.textContent='BLADES: SHEATHED';}});
  ScrollTrigger.create({trigger:'#drop',start:'top center',onEnter(){hudState.textContent='BLADES: ONE FALLS';}});
  ScrollTrigger.create({trigger:'#slash',start:'top center',onEnter(){hudState.textContent='BLADES: ONI GIRI';}});
})();
